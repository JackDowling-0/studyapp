#ifndef FLASHCARDS_H
#define FLASHCARDS_H


#include <string>
#include <vector>
#include <iostream>
#include <algorithm>

namespace Flash 
{
    //TODO - 2. Create public ID (what card are we currently looking at)
    // flashcard type
    class Flashcard
    {

	public:
		//library to hold flash cards
		std::vector<Flashcard> library;

        private:
        // Attributes
            int ID;

            std::string question;
            std::string answer;
        
        
        // Methods
            void updateID(int startingIndex) {
                int i = startingIndex + 1;
                for (auto it = library.begin() + startingIndex; it != library.end(); it++) {
                    it->ID = i;
                    i++;
                }
            }
        
        // Public attributes
        public:

	    bool ValidID = [&](int cardNumber) -> bool {
		    if (cardNumber > 0 && cardNumber <= library.size() + 1){
			    return true;
		    }
		    return false;
	    };
